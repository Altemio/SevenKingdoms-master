is_dragon_rider = {
	is_alive = yes
	trait = dragon_rider
	any_friend = {
		is_tamed_dragon = yes
	}
}

is_basic_dragon = {
	is_alive = yes
	culture = dragon_culture
	religion = dragon_religion
	trait = dragon_trait
	trait = dragon_base_trait
}

is_wild_dragon = {
	is_basic_dragon = yes
	is_tamed_dragon = no
}

is_tamed_dragon = {
	is_basic_dragon = yes
	any_friend = {
		trait = dragon_rider
	}
}
use_dragon_in_battle_potential = {
	in_command = yes
	in_battle = yes
	is_dragon_rider = yes
}

use_dragon_in_siege_potential = {
	in_command = yes
	in_battle = no
	siege = { is_attacker = yes }
	is_dragon_rider = yes
}

dragon_not_ready_for_battle = {
	OR = {
		trait = dragon_wounded
		trait = dragon_maimed
	}
}

killed_by_dragon = {
	OR = {
		death_reason = death_dragon_fire1
		death_reason = death_dragon_fire2
	}
}

is_dragon_in_dragon_duel = {
	OR = {
		has_character_flag = debug_dragon_a
		has_character_flag = debug_dragon_b
		has_character_flag = debug_dragon_c
		has_character_flag = debug_dragon_d
		has_character_flag = debug_dragon_e
		has_character_flag = debug_dragon_f
	}
}

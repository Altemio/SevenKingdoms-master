namespace = dragon

#calculates the dragon's stength
character_event = {
	id = dragon.1

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_dragon = yes
	}
	
	immediate = {
		ROOT = { 
			export_to_variable = { which = comratvar value = combat_rating }
			export_to_variable = { which = healthvar value = health }
			export_to_variable = { which = sizeforlocvar value = martial }
			export_to_variable = { which = sizevar value = martial }
			export_to_variable = { which = drastrvar value = martial }
			divide_variable = { which = sizevar value = 10 }
			divide_variable = { which = drastrvar value = 10 }
			multiply_variable = { which = comratvar which = healthvar }
			change_variable = { which = drastrvar which = comratvar }
			#divide_variable = { which = drastrvar value = 10 }
			if = {
				limit = { is_tamed_dragon = yes }
				set_variable = { wich = dragon_status value = 2 }
			}
		}
		repeat_event = { id = dragon.1 days = 31 }
	}
}

#kills the dragon
character_event = {
	id = dragon.3

	#is_triggered_only = yes
	hide_window = yes

	trigger = {
		is_dragon = yes
		age = 110
	}
	
	mean_time_to_happen = { days = 1 }
	
	immediate = {
		random = {
			chance = 30
			modifier = {
				factor = 0
				age < 110
			}
			modifier = {
				factor = 1.2
				age <= 120
				NOT = { age <= 110 }
			}
			modifier = {
				factor = 1.4
				age <= 130
				NOT = { age <= 120 }
			}
			modifier = {
				factor = 1.6
				age <= 140
				NOT = { age <= 130 }
			}
			modifier = {
				factor = 1.8
				age <= 150
				NOT = { age <= 140 }
			}
			modifier = {
				factor = 2.2
				age <= 160
				NOT = { age <= 150 }
			}
			death = { death_reason = death_natural } 
		}
	}
}

				random_traits = no

					martial = 0
					diplomacy = 0
					stewardship = 0

					learning = 0
				}
				immortal_age = 0
				trait = "dragonegg"
				trait = "dragon_base_trait"
				trait = "dragon_trait"
			}
			new_character = {
				set_father = ROOT
			}
		}
	}
}

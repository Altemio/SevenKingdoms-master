obj_train_dueling = {
	type = character
	
	potential = {
        is_adult = yes
        is_playable = yes
        
        NOR = {
            has_character_flag = improved_combat_skills
            
            age = 40
            trait = dueling_master
			trait = dueling_legend
            
            is_maimed_trigger = yes
            has_severe_disability_trigger = yes
            has_medium_disability_trigger = yes
            has_disease_trigger = yes
        }
    }
    
    allow = {
		is_adult = yes
        is_ruler = yes
        NOT = { trait = incapable }
    }
	
	chance = {
		factor = 100
		modifier = { 
			is_female = yes
			factor = 0.5
		}
	}
    
	success = { 
		has_character_flag = improved_combat_skills
	}
    
	abort = {
		OR = {
			is_maimed_trigger = yes
            has_severe_disability_trigger = yes
            has_medium_disability_trigger = yes
            has_disease_trigger = yes
            
			age = 40
		}
	}
    
	abort_effect = {
	}
    
	effect = {
		prestige = 20
	}
}

    is_government_potential = muslim_government
    is_government_potential = old_gods_government
    is_government_potential = stepstones_government
  }
}

is_freehold_pretender_potential = {
  AND = { 
    tier = EMPEROR
    primary_title = { has_title_flag = pretender_freehold_empire }
    OR = {
      culture_group = valyrian
      has_character_flag = freehold_imperial_government_preserve
    }
  }
}

namespace = dynamic_coa

#	Set CoA
character_event = {
	id = dynamic_coa.0

	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_dynasty_with_special_coa = yes
	}
	
	option = {
		set_dynamic_coa = yes
	}
}

#	Reset CoA if no match on new holder/on death
character_event = {
	id = dynamic_coa.1

	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_dynamic_coa = yes
		is_dynasty_with_special_coa = no
	}
	
	option = {
		any_demesne_title = {
			reset_coa = PREV
			clear_title_flag = is_dynamic_coa
		}
	}
}

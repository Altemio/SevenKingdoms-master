	id = DAWN.1

	}
}

character_event = {
    id = DAWN.3
    desc = "EVTDESCDAWN.3"
    picture = GFX_sword_of_the_morning
    major = yes
    is_triggered_only = yes

   
    only_playable = yes
   
    trigger = {
        NOT = {
            has_global_flag = DAWN_event_3
        }
    }
    major_trigger = {

            has_artifact = dawn

    }

                }
            }

    option = {
        name = "EVTOPTBDAWN.3"
        ai_chance = {
            factor = 1
            modifier = {
                factor = 1
                OR = {
                    trait = cruel

                    trait = dishonorable
                }
            }
        }
    }
    after = {

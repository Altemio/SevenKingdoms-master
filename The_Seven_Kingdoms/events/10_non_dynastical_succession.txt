## Non-dynastical succession ##

character_event = {
	id = nds.0

	is_triggered_only = yes
	hide_window = yes
	
	trigger = { #the trigger is taken from emf_core
		has_game_rule = { name = non_dynastic_succession value = on }
		is_nomadic = no
		is_merchant_republic = no
		ai = no
		OR = {
			capital_scope = {
				county = {
					holder = ROOT
					current_heir = {
						ai = yes
						NOT = { dynasty = ROOT }
					}
				}
			}
			AND = {
				OR = {
					mercenary = yes
					holy_order = yes
				}
				current_heir = {
					ai = yes
					NOT = { dynasty = ROOT }
				}
			}
		}
	}
	
	immediate = {
		current_heir = { 
			ROOT = { set_player_character = PREV }
		}
	}
	
	option = { }
}

###Dueling Events###

#Player seeks to train
namespace = dueling
character_event = {
	id = dueling.1
	desc = "EVTDESCdueling.1"
	
	hide_from = yes
	
	is_triggered_only = yes
	
	min_age = 16
	capable_only = yes
	prisoner = no
	
	trigger = {
        NOT = {
            has_character_flag = already_trained
        }
    }
	
	option = {
		name = "EVTOPTAdueling.1" #I shall become stronger
		ai_chance = { factor = 100 }
		set_character_flag = already_trained
		if = {
			limit = { trait = dueling_novice }
			random = { chance = 80
				hidden_tooltip = { remove_trait = dueling_novice }
				add_trait = dueling_adept
			}
			random = { chance = 20
				add_trait = wounded
			}
			break = yes
		}
		if = {
			limit = { trait = dueling_adept }
			random = { chance = 50
				hidden_tooltip = { remove_trait = dueling_adept }
				add_trait = dueling_skilled
			}
			random = { chance = 20
				add_trait = wounded
			}
			break = yes
		}
		if = {
			limit = { trait = dueling_skilled }
			random = { chance = 30
				hidden_tooltip = { remove_trait = dueling_skilled }
				add_trait = dueling_expert
			}
			random = { chance = 20
				add_trait = wounded
			}
			break = yes
		}
		if = {
			limit = { trait = dueling_expert }
			random = { chance = 10
				hidden_tooltip = { remove_trait = dueling_expert }
				add_trait = dueling_master
			}
			random = { chance = 20
				add_trait = wounded
			}
			break = yes
		}
		if = {
			limit = { trait = dueling_master }
			random = { chance = 1
				hidden_tooltip = { remove_trait = dueling_master }
				add_trait = dueling_legend
			}
			random = { chance = 20
				add_trait = wounded
			}
			break = yes
		}
		if = {
			limit = { 
				NOT = {
					OR = {
						trait = dueling_novice
						trait = dueling_adept
						trait = dueling_skilled
						trait = dueling_expert
						trait = dueling_master
						trait = dueling_legend
					}
				}
			}
			add_trait = dueling_novice
		}
	}
}

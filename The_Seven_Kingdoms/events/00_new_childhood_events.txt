# on_adolescence pulse, firing at 12 years plus six months and 14 years plus six months
# Instead of using events that may or may not fire, I will assure that the event fires at least twice
# and then manually adjust the chance of getting traits through one event.

#! Childhood Strength/Weakness Event Chain Start

character_event = {
	id = NCE.15998
	desc = "EVTDESC15998"
	picture = "GFX_evt_child_play"
	
	is_triggered_only = yes
	
	only_capable = yes
    
	trigger = {
		
        NOR = {
			trait = incapable
			is_inaccessible_trigger = yes
			has_character_modifier = the_dungeon
			has_character_modifier = the_oubliette
		}
        
		age = 12
		NOT = { age = 16 }
        
		educator = {
			always = yes
		}
	}
    
    # The effects
    immediate = {
        
        random_list = {
            # A chance of getting strong, improved by being fit, reduced by being weak
            # Stuff that lowers the chance starts off earlier on, with chance increasing stuff later.
            # So lots of negatives have less of an impact as they stack. I think.
            
            # 30 chance of no effect
            # 10 Becomes strong
            # 15 Becomes fit
            # 15 Becomes weak
            # 20 Weak becomes no trait
            
            50 = {                      # No effect at all
                 modifier = {
                    factor = 1.2          # Higher chance of no effect, fit
                    trait = fit
                }
                modifier = {
                    factor = 0.8          # Slightly lower chance of no effect, weak
                    trait = weak
                }
            }
            
            10 = {
                modifier = {
                    factor = 0          # Since we already have the trait, reduction
                    trait = strong
                }
                modifier = {            # Being of a poor physique just isn't compatible with being strong.
                    factor = 0
                    trait = poor_physique
                }
                
                modifier = {
                    factor = 0.3        # Our character is weak, so this would be a big jump. Possible, but rare.
                    trait = weak
                }
                
                modifier = {
                    factor = 0.9
                    trait = lisp
                }
                modifier = {
                    factor = 0.9
                    trait = stutter
                }
                modifier = {
                    factor = 0.9
                    trait = harelip
                }
                modifier = {
                    factor = 0.7
                    trait = hunchback
                }
                modifier = {
                    factor = 0.6
                    trait = clubfooted
                }
                modifier = {
                    factor = 0.5
                    trait = inbred
                }
                
                modifier = {            # Characters who are both content and fit are happy the way they are.
                    factor = 0.5
                    trait = content
                    trait = fit
                }
                
                modifier = {            # Over eating and lazy makes a great pair for being weak.
                    factor = 0.3
                    trait = gluttonous
                    trait = slothful
                }
                
                modifier = {            # Certain personalities and trait combinations make them more likely to be lazy
                    factor = 0.7
                    OR = {
                        trait = chaste
                        trait = humble
                        trait = craven
                    }
                }
                
                modifier = {            # A sharp mind can make the body soft
                    factor = 0.8
                    OR = {
                        trait = genius
                        trait = intelligent
                    }
                }
                
                modifier = {            # Others do not lend their way to physical activity
                    factor = 0.8
                    OR = {
                        trait = celibate
                        trait = scholar
                        trait = mystic
                        trait = poet
                    }
                }
                
                modifier = {            # Assortment of positive traits
                    factor = 1.1
                    OR = {
                        trait = lustful
                        trait = temperate
                        trait = greedy
                        trait = diligent
                        trait = envious
                        trait = wroth
                        trait = patient
                        trait = proud
                        trait = brave
                        trait = ambitious
                    }
                }
                
                modifier = {            # Lifestyle traits may result in fitness
                    factor = 1.2
                    OR = {
                        trait = hedonist
                        trait = gardener
                        trait = impaler
                        trait = duelist
                        trait = hunter
                        trait = falconer
                    }
                }
                
                modifier = {            # Being big certainly helps with strength, due to the attention it attracts
                    factor = 1.2
                    trait = huge
                }
                
                modifier = {
                    factor = 1.3        # If our character is fit, they have a greater chance of becoming strong.
                    trait = fit
                }
                
                modifier = {
                    factor = 1.5        # Already fit and huge? Much greater chance!
                    trait = fit
                    trait = huge
                }
                
                modifier = {
                    factor = 2.0        # If our character has a strong physique, they are double as likely to become strong.
                    trait = strong_physique
                }
                
                modifier = {
                    factor = 1.3        # Ambitious makes them want to improve, but only if proud or envious AND not humble
                    trait = ambitious
                    trait = fit
                    OR = {
                        trait = proud
                        trait = envious
                    }
                    NOT = {
                        trait = humble
                    }
                }
                
                # Removed other traits, add strong
                remove_trait = fit
                remove_trait = weak
                add_trait = strong
                
            }
            
            # Effects of negative traits are much less drastic for fit, but still apparent
            
            10 = {
                modifier = {
                    factor = 0           # Since we have a better version, negate
                    trait = strong
                }
                modifier = {
                    factor = 0          # Since we already have the trait, negate
                    trait = fit
                }
                
                modifier = {            # Being of a poor physique makes it much more difficult to be fit.
                    factor = 0.5
                    trait = poor_physique
                }
                
                modifier = {
                    factor = 0.7        # Reduced chance if the character is already weak
                    trait = weak
                }
                
                modifier = {
                    trait = hunchback
                    factor = 0.8
                }
                modifier = {
                    trait = clubfooted
                    factor = 0.7
                }
                modifier = {
                    trait = inbred
                    factor = 0.6
                }
                
                modifier = {            # Characters who are both content and trait less are less likely to improve.
                    factor = 0.8
                    trait = content
                    NOT = {
                        trait = strong
                        trait = fit
                        trait = weak
                    }
                }
                
                modifier = {
                    factor = 2.0        # If our character has a strong physique, they are double as likely to become strong.
                    trait = strong_physique
                }
                
                modifier = {
                    factor = 1.2        # Ambitious makes them want to improve, but only if proud or envious AND not humble
                    trait = ambitious
                    OR = {
                        trait = proud
                        trait = envious
                    }
                    NOT = {
                        trait = humble
                    }
                }
                
                modifier = {            # Lifestyle traits may result in fitness
                    factor = 1.2
                    OR = {
                        trait = hedonist
                        trait = gardener
                        trait = impaler
                        trait = duelist
                        trait = hunter
                        trait = falconer
                    }
                }
                
                modifier = {            # Assortment of positive traits
                    factor = 1.2
                    OR = {
                        trait = lustful
                        trait = temperate
                        trait = greedy
                        trait = diligent
                        trait = envious
                        trait = wroth
                        trait = patient
                        trait = proud
                        trait = brave
                        trait = ambitious
                    }
                }
                
                modifier = {            # Others do not lend their way to physical activity
                    factor = 0.9
                    OR = {
                        trait = celibate
                        trait = scholar
                        trait = mystic
                        trait = poet
                    }
                }
                
                remove_trait = strong
                remove_trait = weak
                add_trait = fit
            }
            
            15 = {
                modifier = {
                    factor = 0          # Since we already have the trait, negate
                    trait = weak
                }
                modifier = {
                    factor = 0.0        # Strong physique means... Strong.
                    trait = strong_physique
                }
                
                modifier = {            # Being of a poor physique makes it more likely that they will be weak.
                    factor = 1.5
                    trait = poor_physique
                }
                
                modifier = {
                    trait = hunchback
                    factor = 1.2
                }
                modifier = {
                    trait = clubfooted
                    factor = 1.3
                }
                modifier = {
                    trait = inbred
                    factor = 1.5
                } 
                
                modifier = {            # Some traits just do not lend their way to physical activity
                    factor = 1.1
                    OR = {
                        trait = celibate
                        trait = scholar
                        trait = mystic
                        trait = poet
                    }
                }
                
                modifier = {            # Certain personalities and trait combinations make them more likely to be lazy
                    factor = 1.2
                    OR = {
                        trait = chaste
                        trait = humble
                        trait = craven
                    }
                }
                
                modifier = {            # A sharp mind can make the body soft
                    factor = 1.1
                    OR = {
                        trait = genius
                        trait = intelligent
                    }
                }
                
                modifier = {            # Over eating and lazy makes a great pair for being weak.
                    factor = 1.5
                    trait = gluttonous
                    trait = slothful
                }
                
                remove_trait = strong
                remove_trait = fit
                add_trait = weak
            }
            
            15 = {                      # Remove weak, no traits
                modifier = {
                    factor = 0          # Don't fire for anyone other than someone who is weak.
                    NOT = {
                        trait = weak
                    }
                }
                modifier = {
                    factor = 1.2        # Ambitious makes them want to improve, but only if proud or envious AND not humble
                    trait = ambitious
                    OR = {
                        trait = proud
                        trait = envious
                    }
                    NOT = {
                        trait = humble
                    }
                }
                modifier = {
                    factor = 1.5        # Strong physique means... Strong.
                    trait = strong_physique
                }
                
                modifier = {            # Characters who are both content and weak put up with the way they are.
                    factor = 0.9
                    trait = content
                    trait = weak
                }
                
                remove_trait = strong
                remove_trait = fit
                remove_trait = weak
            }
        }
    }
    
    option = {
		name = "EVTOPTA15998"
        
        name = {
          trigger = {
            trait = strong
          }
          text = EVTDESC_CASE115998
        }

        name = {
          trigger = {
            trait = fit
          }
          text = EVTDESC_CASE215998
        }
        
        name = {
          trigger = {
            trait = weak
          }
          text = EVTDESC_CASE315998
        }
        
        name = {
          trigger = {
            NOR = {
                trait = strong
                trait = weak
                trait = weak
            }
          }
          text = EVTDESC_CASE415998
        }

            
        # Notification for Guardian
        hidden_tooltip = {
            guardian = {
                character_event = {
                    id = NCE.15999
                    days = 3
                }
            }
        }
	}
}

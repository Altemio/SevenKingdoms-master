# -*- ck2.events -*-

namespace = trhk_ambition

character_event = {
	id = trhk_ambition.1

	trigger = {
		ai = yes
		prestige = 1000
		war = no
		NOT = { 
			has_character_modifier = high_kingdom_ambition
		}
		OR = {
			has_ambition = obj_become_high_king_north
			has_ambition = obj_become_high_king_riverlands
			has_ambition = obj_become_high_king_westerlands
			has_ambition = obj_become_high_king_dusklands
			has_ambition = obj_become_high_king_dorne
		}
	}
	
	option = {
		ai_chance = { factor = 70 }
		
		if = {
			limit = { has_ambition = obj_become_high_king_north }
			random_neighbor_independent_ruler = {
				limit = { 
					capital_scope = { region = world_the_north } 
					NOT = { is_tributary = { type = permanent suzerain = ROOT } }
				}
				reverse_war = { 
					target = PREV 
					casus_belli = high_kingdom_subjugation 
				}
				add_character_modifier = { name = high_kingdom_ambition  duration = 274 }
			}
		}
		else_if = {
			limit = { has_ambition = obj_become_high_king_riverlands }
			random_neighbor_independent_ruler = {
				limit = { 
					capital_scope = { region = world_the_riverlands } 
					NOT = { is_tributary = { type = permanent suzerain = ROOT } }
				}
				reverse_war = { 
					target = PREV 
					casus_belli = high_kingdom_subjugation 
				}
				add_character_modifier = { name = high_kingdom_ambition  duration = 274 }
			}
		}
		else_if = {
			limit = { has_ambition = obj_become_high_king_westerlands }
			random_neighbor_independent_ruler = {
				limit = { 
					capital_scope = { region = world_westerlands } 
					NOT = { is_tributary = { type = permanent suzerain = ROOT } }
				}
				reverse_war = { 
					target = PREV 
					casus_belli = high_kingdom_subjugation 
				}
				add_character_modifier = { name = high_kingdom_ambition  duration = 274 }
			}
		}
		else_if = {
			limit = { has_ambition = obj_become_high_king_dusklands }
			random_neighbor_independent_ruler = {
				limit = { 
					capital_scope = { region = world_the_blackwater } 
					NOT = { is_tributary = { type = permanent suzerain = ROOT } }
				}
				reverse_war = { 
					target = PREV 
					casus_belli = high_kingdom_subjugation 
				}
				add_character_modifier = { name = high_kingdom_ambition  duration = 274 }
			}
		}
		else_if = {
			limit = { has_ambition = obj_become_high_king_dorne }
			random_neighbor_independent_ruler = {
				limit = { 
					capital_scope = { region = world_dorne } 
					NOT = { is_tributary = { type = permanent suzerain = ROOT } }
				}
				reverse_war = { 
					target = PREV 
					casus_belli = high_kingdom_subjugation 
				}
				add_character_modifier = { name = high_kingdom_ambition  duration = 274 }
			}
		}
	}
	
	option = {
		ai_chance = { factor = 30 }
	}
}

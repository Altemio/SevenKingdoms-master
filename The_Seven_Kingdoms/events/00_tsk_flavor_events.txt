# Mountain Clansmen Ambush 
character_event = {
	id = tsk.0104
	desc = EVTDESCtsk.0104
	picture = GFX_evt_arryn2
	border = GFX_event_normal_frame_religion
	
	min_age = 16
	only_rulers = yes
	only_capable = yes
	prisoner = no
	
	trigger = {
		is_landed = yes
		is_abroad = no
		OR = {
			has_landed_title = e_the_vale
			religion_group = the_seven_group
		}
	}
	
	mean_time_to_happen = {
		months = 400
		modifier = {
			factor = 0.50
			trait = zealous
		}
		modifier = {
			factor = 0.25
			trait = wroth
		}
		modifier = {
			factor = 0.25
			trait = cruel
		}
	}
	
	option = {
		name = EVTOPTAtsk.0104 # I will hunt them down
		ai_chance = {
			factor = 50
		}
		prestige = 50
		custom_tooltip = {
			text = hunt_them_down
			hidden_tooltip = {
				character_event = { id = tsk.0105 days = 5 }
			}
		}
	}
	option = {
		name = EVTOPTBtsk.0104 # That sounds dangerous
		ai_chance = {
			factor = 50
		}
		prestige = -50
		add_trait = craven
	}
}

					character_event = { id = tsk.0107 } # You are killed in battle
					}
				}
			}
		}
	}
}

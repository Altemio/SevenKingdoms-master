character_event = {
	id = TSK_MARKET.1
	desc = {
		trigger { NOT = has_character_flag = looped_market }
		text = EVTDESCA_TSK_MARKET1
	}
	desc = { 
		trigger { has_character_flag = looped_market }
		text = EVTDESCB_TSK_MARKET1
		}
	# picture =
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA_TSK_MARKET1 #Takes you to the Weaponsmith (ID 10)
		character_event = { id = TSK_MARKET.10 days = 3 tooltip = EVTTOOLTIP_TSK_MARKET1A }
	}
	option = {
		name = EVTOPTB_TSK_MARKET1 #Takes you to the Armorsmith (ID 20)
		character_event = { id = TSK_MARKET.20 days = 3 tooltip = EVTTOOLTIP_TSK_MARKET1B }
		
	}	# Go to the weaponsmith
	option = {
		name = EVTOPTC_TSK_MARKET1 #Takes you to the Jeweler (ID 30)
		character_event = { id = TSK_MARKET.30 days = 3 tooltip = EVTTOOLTIP_TSK_MARKET1C }
		
	}	
	option = {
		name = EVTOPTD_TSK_MARKET1 #Cycle through the options (ID 2)
		character_event = { id = TSK_MARKET.2 days = 3 tooltip = EVTTOOLTIP_TSK_MARKET1D }
	}
}
